
# PSM API  

# Requirement:
    Node : 16.17 
    PORT : 4444
# Step For Run Application : 
    - npm i 
    - npm run build 
    - NODE_ENV=dev npm start & - for run background 

# Step  -  for use pm2 
    - sudo npm install pm2 -g
    - npm run build 
    - NODE_ENV=dev npm run pm2 

# Step - For use Docker 
    Requirement :  docker 
        1. docker-compose down
        2. docker-compose build
        3. docker-compose up
